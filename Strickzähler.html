<h2>Reihen seit letzter Zunahme</h2>

<div id="reihe" class="number">0</div>


<h2>Zunahmen insgesamt</h2>

<div id="zunahmen" class="number">0</div>


<button onclick="incRow()">+1 Reihe</button><br>

<button onclick="resetAll()">Alles zurücksetzen</button>


<div id="hinweis"></div>


<script>

const LIMIT = 8;


// Werte aus Speicher laden

let reihe = parseInt(localStorage.getItem("reihe")) || 0;

let zunahmen = parseInt(localStorage.getItem("zunahmen")) || 0;


function update() {

    document.getElementById("reihe").textContent = reihe;

    document.getElementById("zunahmen").textContent = zunahmen;


    // speichern

    localStorage.setItem("reihe", reihe);

    localStorage.setItem("zunahmen", zunahmen);


    // Hinweis bei Reihe 7

    const hinweis = document.getElementById("hinweis");

    if (reihe === 7) {

        hinweis.textContent = "Jetzt Zunahme stricken!";

    } else {

        hinweis.textContent = "";

    }

}


function incRow() {

    reihe++;


    // automatische Zunahme bei Reihe 8

    if (reihe >= LIMIT) {

        zunahmen++;

        reihe = 0;

    }


    update();

}


function resetAll() {

    if (confirm("Alles zurücksetzen?")) {

        reihe = 0;

        zunahmen = 0;

        update();

    }

}


update();

</script>
